```tsx
import React, { useState, useEffect, useRef } from 'react';
import { motion, useMotionValue, useTransform } from 'framer-motion';

const RoomCanvas: React.FC = () => {
  const [gridSize, setGridSize] = useState(40); // 그리드 셀 크기
  const [objectPosition, setObjectPosition] = useState({ x: 0, y: 0 });
  const [isDragging, setIsDragging] = useState(false);
  const canvasRef = useRef<HTMLDivElement>(null);

  // 아이소메트릭 좌표 변환 함수
  const isoToScreen = (x: number, y: number): { top: number, left: number } => {
    const cellWidth = gridSize * 2;
    const cellHeight = gridSize * 2;
    return {
      top: (y * cellHeight / 2),
      left: (x * cellWidth - y * cellWidth)
    };
  };

  // 마우스 이벤트 핸들러
  const handleMouseDown = (e: React.MouseEvent) => {
    setIsDragging(true);
  };

  const handleMouseMove = (e: React.MouseEvent) => {
    if (!isDragging) return;
    const rect = canvasRef.current?.getBoundingClientRect();
    if (!rect) return;

    // 마우스 위치 -> 그리드 좌표 변환
    const mouseX = e.clientX - rect.left;
    const mouseY = e.clientY - rect.top;

    const gridX = Math.round(mouseX / (gridSize * 2));
    const gridY = Math.round(mouseY / (gridSize * 1.5));

    setObjectPosition({ x: gridX, y: gridY });
  };

  const handleMouseUp = () => {
    setIsDragging(false);
  };

  // 그리드 렌더링
  const renderGrid = () => {
    const rows = 10;
    const cols = 15;
    return Array.from({ length: rows }, (_, row) =>
      Array.from({ length: cols }, (_, col) => {
        const { top, left } = isoToScreen(col, row);
        return (
          <div
            key={`${row}-${col}`}
            className="absolute w-4 h-4 bg-gray-300 opacity-20"
            style={{ top, left }}
          />
        );
      })
    );
  };

  return (
    <div
      ref={canvasRef}
      className="w-full h-screen relative cursor-move"
      onMouseDown={handleMouseDown}
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
    >
      {/* 아이소메트릭 그리드 */}
      <div className="absolute inset-0 pointer-events-none">
        {renderGrid()}
      </div>

      {/* 드래그 가능한 오브젝트 */}
      <motion.div
        className="absolute w-12 h-12 bg-blue-500 rounded"
        style={{
          top: useTransform(objectPosition.y, (y) => isoToScreen(objectPosition.x, y).top),
          left: useTransform(objectPosition.x, (x) => isoToScreen(x, objectPosition.y).left)
        }}
        animate={{
          x: useTransform(objectPosition.x, (x) => isoToScreen(x, objectPosition.y).left),
          y: useTransform(objectPosition.y, (y) => isoToScreen(objectPosition.x, y).top)
        }}
        transition={{ type: "spring", stiffness: 300, damping: 20 }}
      />
    </div>
  );
};

export default RoomCanvas;
```

### 🎨 1. 핵심 감성 방향성 (The Vibe)

| 요소 | 설명 | 목표 감성 | 구현 방법 |
| :--- | :--- | :--- | :--- |
| **빛의 연출 (Glow)** | 오브젝트와 선택된 그리드 포인트 주변에 부드럽게 번지는 **Bloom 효과**를 적용합니다. 하드한 경계선을 없애고, 디지털 오브젝트에 '생명력'을 부여합니다. | 마법적, 생동감 있는 (Magical, Alive) | CSS `box-shadow` (반투명, 큰 blur 값) |
| **색상 (Palette)** | 채도가 높기보다는 **뮤트(Muted)된 파스텔 톤**의 그라디언트를 사용합니다. 기본 배경은 아늑하고 따뜻한 느낌을 유지합니다. | 아늑함, 창의적 (Cozy, Creative) | Soft Gradient (예: Sky Blue to Lavender) |
| **애니메이션 (Motion)** | 빠르고 경직된 움직임 대신, **물리학 기반의 탄성(Elastic) 효과**를 적용합니다. 오브젝트가 제자리를 찾을 때 살짝 튕기는 느낌을 줍니다. | 만족감, 직관적 (Satisfying, Intuitive) | Framer Motion의 `spring` 또는 `stiffness` 속성 활용 |

### 📐 2. 상세 디자인 요소별 가이드

#### 🔵 A. 배경 및 그리드 시스템 (The Foundation)

*   **배경색 (Canvas Background):** `#FAF9F6` (Soft Ivory). 차가운 흰색 대신 따뜻한 느낌을 주어 사용자가 편안하게 장시간 작업할 수 있게 합니다.
*   **그리드 라인 (Grid Line):** `#E0E0E0` (거의 보이지 않는 회색). 선명한 검정색 대신 매우 연한 회색을 사용해, 그리드가 '보조 도구'임을 인지시키고 오브젝트가 주인공이 되게 합니다.
*   **스냅 포인트 강조 (Snap Highlight):** 오브젝트가 스냅될 때, 그리드 라인 자체가 아닌 **스냅 포인트 주변에만** `#A8AFFF` (Soft Lavender Glow)가 아주 잠깐 깜빡이는 효과를 줍니다.

#### 🟠 B. 오브젝트 배치 및 구조 (The Objects)

*   **오브젝트 기본 형태:** 각 오브젝트는 단순한 사각형이 아닌, **모서리가 둥글게 처리된(Rounded Corners)** 카드형 또는 캡슐형 디자인을 기본으로 합니다. (UX: 딱딱함을 제거하고 부드러움을 더함).
*   **오브젝트 라이브러리 (Source):** 라이브러리에서 오브젝트를 선택할 때, 해당 오브젝트의 미리보기 이미지는 **Soft Gradient**가 적용된 3D 카드 형태로 보여줍니다. (Hover 시 미세한 빛 반사 효과 추가).
*   **레이어링 (Depth):** 오브젝트가 겹칠 때, 전경 오브젝트는 약간의 **반투명 효과(Opacity 90%)**를 주어 깊이감을 느끼게 하고, 배경 오브젝트는 더 흐리게(Blur) 처리하여 시선이 전경에 집중되도록 유도합니다.

#### 🔴 C. 애니메이션의 '느낌' (The Motion)

| 액션 (Action) | 애니메이션 느낌 (Feeling) | 기술 구현 (Implementation) | UX 기여 (UX Contribution) |
| :--- | :--- | :--- | :--- |
| **드래그 & 스냅** | **"찰싹! (Snap & Bounce)"** 오브젝트를 이동시키다가 그리드에 닿을 때, 목표 지점에서 살짝 **과도하게 튕겼다가(Overshoot)** 부드럽게 멈추는 탄성 애니메이션. | `stiffness`와 `damping`이 적용된 Spring Motion (0.3s) | *만족감 극대화.* "제자리에 정확히 안착했다"는 시각적 확신을 제공합니다. |
| **생성/드롭** | **"떠오름 (Soft Fade-In)"** 캔버스에 오브젝트가 놓이는 순간, 0.1초 동안 투명도 0%에서 100%로 부드럽게 올라오면서, 미세한 **Bloom Glow**가 퍼지며 나타납니다. | Opacity Transition (0.1s) + Glow Shadow | *환상적 경험.* 오브젝트가 '생성'되는 마법 같은 순간을 연출합니다. |
| **선택/활성화** | **"반짝임 (Pulse Highlight)"** 오브젝트를 클릭하거나 선택할 때, 오브젝트 전체가 0.2초 동안 **은은한 빛의 펄스(Pulse)**를 내뿜으며 테두리가 강조됩니다. | Keyframe Animation (Pulse effect) | *사용자 피드백.* 현재 사용자가 무엇을 조작하고 있는지 명확하게 인지시켜줍니다. |

### 💡 최종 사용자 경험(UX) 기여 요약

이러한 'Dream Weaver' 디자인 방향성은 다음 세 가지 핵심 UX 문제를 해결하여, MVP의 완성도를 극대화합니다.

1.  **인지 부하 감소 (Reduced Cognitive Load):** 차분한 배경색과 은은한 그리드 라인은 사용자가 '이 캔버스를 써야 한다'는 부담감을 느끼지 않고, 자유롭게 아이디어를 배치하도록 돕습니다.
2.  **만족감 극대화 (Sense of Accomplishment):** 탄성 애니메이션과 스냅 효과는 단순한 기능 작동을 '성취감' 있는 경험으로 전환시킵니다. (사용자가 "와, 이거 진짜 재미있다"라고 느끼게 만듭니다.)
3.  **프로페셔널함 유지 (Maintaining Polish):** Bloom Glow와 부드러운 그라디언트 사용은 툴이 '취미용'이 아닌, 고도로 세련된 '디자인 전문 도구'처럼 보이게 합니다.
